def create
  autorize_new
  @article = Article.new(article_params)
  @article.user = current_user
  respond_to do |format|
    if @article.save
      format.html { redirect_to edit_article_path(@article), notice: 'Article was successfully created.' }
      format.json { render :show, status: :created, location: @article }
    else
      format.html { render :new }
      format.json { render json: @article.errors, status: :unprocessable_entity }
    end
  end
end

def related_articles
  @related_articles = Article
  .translation_published(I18n.locale)
  .joins(:taggings)
  .where('articles.id != ?', @article.id)
  .where(taggings: { tag_id: @article.tag_ids })
  .where(area_id: @article.area_id)
  .group("articles.id")
  .limit(6)
end

def set_article_draft
  translations = ActiveSupport::JSON.decode(params[:contents])
  translated = {}
  translations.each do |key, array|
    if key == params[:draft]
      translated[key] = "draft"
    end
  end
  @article.is_translated = translated
end

def show
  impressionist(@article, "view article",:unique => [:session_hash])
  self.related_articles
  @page = params.has_key?(:page) ?  params[:page] : 1
  @article.get_i18n(I18n.locale)
  if @article.category == nil
    add_breadcrumb "Articles", "/"
  else
    add_breadcrumb @article.category.name, categoryListing_path(@article.category)
  end
  add_breadcrumb @article.disp_title[0], "#{@article.slug}"
  articlePath = params.has_key?(:page) ? article_path(@article,params[:page]) : article_path(@article)
  if request.path != articlePath
    return redirect_to articlePath, status: :moved_permanently
  else
    respond_to do |format|
      format.html # show.html.erb
      format.json { render json: @article }
    end
  end
end

def update
  if(params.has_key?(:draft))
    self.set_article_draft
  else
    self.set_article_translated
  end
  respond_to do |format|
    @article.title = ActiveSupport::JSON.decode(params[:disp_title])["en"]
    @article.slug = nil
    @article.publish
    if @article.update(article_params)
      #format.html { redirect_to @article, notice: 'Article was successfully updated.' }
      format.json { render :show, status: :ok, location: @article }
    else
      format.html { render :edit }
      format.json { render json: @article.errors, status: :unprocessable_entity }
    end
  end
end
