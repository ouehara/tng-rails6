def create
  @pickup = Pickup.new(pickup_params)

  respond_to do |format|
    if @pickup.save
      format.html { redirect_to [:admin, @pickup], notice: 'Pickup was successfully created.' }
    else
      format.html { render :new }
    end
  end
end
