def list_articles
   page = 1
    if(params.has_key?(:page))
      page = params[:page]
    end
    @title = t('tags.title',param:params[:tag])
    @desc = t('tags.desc')
    add_breadcrumb params[:tag], :tag_path
    @articles = Article.tagged_with(params[:tag]).translation_published(I18n.locale).page(page).per(Rails.configuration.elements_per_page).order("published_at desc")
    render "articles/index"
end
