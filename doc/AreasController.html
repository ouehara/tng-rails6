def list_articles
    page = 1
    if(params.has_key?(:page))
      page = params[:page]
    end
    area = Area.find(params[:area])
    @title = t('area.title',param: area.name)
    @desc = t('area.desc')
    @curArea = area
    add_breadcrumb "Areas", :area_listing_path
    @articles = Article.find_by_area(params[:area]).translation_published(I18n.locale).page(page).per(Rails.configuration.elements_per_page).order("published_at desc")
    add_breadcrumb area.name, :area_path
    render "articles/index"
end
