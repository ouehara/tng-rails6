def current_translations
  I18n.backend.send(:init_translations) unless I18n.backend.initialized?
  translations ||= I18n.backend.send(:translations)
  @translations = translations[I18n.locale]
end

def setup_sidebar
  articles = Article.translation_published(I18n.locale).order(impressions_count: :desc, id: :desc).limit(5)
  tags = Tag
        .joins("left join taggings on tags.id = taggings.tag_id")
        .group("tags.id")
        .order("COUNT(tags.id) DESC")
        .includes(:translations)
        .limit(40)
  areas = Area.includes(:translations).where(in_sidebar: true)
  pickups = Pickup.published 
  @sidebar = {topArticles: articles.to_json(:methods => [:square_url]),tags: tags, areas: areas, pickup: pickups.to_json(:methods => [:medium_url])}
end
